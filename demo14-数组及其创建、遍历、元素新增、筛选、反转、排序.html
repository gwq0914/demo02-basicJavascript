数组是指一组数据的集合，其中的每个数据被称作元素，在数组中可以存放任意类型的元素，数组是一种将一组数据存储在单个变量名下的优雅方式。数组可以把一组相关的数据一起存放，并提供方便的访问（获取）方式。
js中创建数组有两种方式：
利用new创建数组；
1.利用数组字面量创建数组；
var 数组名 = [];
2.使用数组字面量方式创建带初始值的数组
var 数组名 = ['red','green','blue'];
数组的索引 ：用来访问数组元素的序号，从0开始。
没有数组元素，输出undefined

访问数组元素案例：一个星期的数组

var week=new Array();
week = ['星期一'，'星期二','星期三','星期四','星期五','星期六','星期日']; console.log(week);
遍历数组
遍历：把数组中的每个元素从头到尾都访问一次
数组的长度：数组名.length，可以访问数组元素的数量，动态监测数组元素的个数

遍历数组案例：数组求和及平均值

题面：求数组[2,6,1,7,4]里面所有元素的和以及平均值
题解：
var num=[2,6,1,7,4],sum=0;
for (var i = 0; i < num.length; i++){
sum +=num[i];
}
console.log("数组[2,6,1,7,4]的总数和是" + sum);
average = sum / num.length;
console.log("数组[2,6,1,7,4]的平均值是" + average);
题面：求数组[2,6,1,7,4]里面所有元素的和以及平均值
题解：
var num=[2,6,1,77,52,25,7],max;
for (var i = 0; i < num.length; i++){
if(num[i]>num[i+1]){
max = num[i];
num[i] = num[i + 1];
num[i + 1] = max;
}
}
console.log("数组[2,6,1,77,52,25,7]的最大值是" + max);

遍历数组案例：数组转换为分割字符串

题面：
将数组['red','green','blue','pink']转换为字符串，并且用|或其他符号分割。
解题：
var color = ["red", "green", "blue", "pink", "black", "white"];
var str = "";
for (var i = 0; i < color.length; i++) {
str += color[i].toString();
if (i == 5) {
continue;
}
str += "|";
}
console.log(str);
数组中新增元素，可以通过修改length长度，索引号新增加数组元素

新增数组案例：

题面：
新建一个数组，里面存放10个数组整数1~10
解题：
var numArr_1 = new Array();
numArr_1.length = 10;
for (var i = 0; i < numArr_1.length; i++) {
numArr_1[i] = i + 1;
};
console.log(numArr_1);

筛选数组案例：

题面：
将数组[2,0,6,1,77,0,52,0,25,7]中大于等于10的元素选出来，放入新数组
题解：
var numArr_2 = [2, 0, 6, 1, 77, 0, 52, 0, 25, 7];
var numArr_3 = [],
j = 0;
for (var i = 0; i < numArr_2.length; i++) {
if (numArr_2[i] >= 10) {
numArr_3[j] = numArr_2[i];
j++;
}
}
console.log(numArr_3);


var numArr_2 = [2, 0, 6, 1, 77, 0, 52, 0, 25, 7];
var numArr_3 = [];
for (var i = 0; i < numArr_2.length; i++) {
if (numArr_2[i] >= 10) {
numArr_3[numArr_3.length] = numArr_2[i];
//新数组中每最后一个元素的索引号 是 新数组上一次元素的长度
//新数组的长度 是 下一次赋值新数组元素的索引号
}
}
console.log(numArr_3);

筛选数组的案例：删除0元素

题面：
要求将数组[2,0,6,1,77,0,52,0,25,7]中的0去掉后，形成一个不包含0的新数组
var numArr_4 = [2, 0, 6, 1, 77, 0, 52, 0, 25, 7];
for (var i = 0; i < numArr_4.length; i++) {
if (numArr_4[i] == 0) {
for (var j = i; j < numArr_4.length; j++) {
numArr_4[j] = numArr_4[j + 1];
}
numArr_4.length--;
}
}
console.log( "[2,0,6,1,77,0,52,0,25,7]中的0去掉后，形成一个不包含0的新数组是：" + numArr_4 );

numArr_4 = [2, 0, 6, 1, 77, 0, 52, 0, 25, 7];
var numArr_5 = [];
for (var i = 0; i < numArr_4.length; i++) {
if (numArr_4[i] != 0) {
numArr_5[numArr_5.length] = numArr_4[i];
}
}
console.log( "[2,0,6,1,77,0,52,0,25,7]中的0去掉后，形成一个不包含0的新数组是：" + numArr_5 );

筛选数组的案例：翻转数组

题面：要求将数组color = ["red", "green", "blue", "orange", "black"];的内容反过来存放，输出color = ["black", "orange", "blue", "green", "red"];
解题：
var numArr_6 = [];
for (var i = color.length - 1; i >= 0; i--) {
numArr_6[numArr_6.length] = color[i];
}
console.log("color的翻转数组：" + numArr_6);
